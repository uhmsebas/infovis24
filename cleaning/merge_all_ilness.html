#change the country names to the standard names
def standardize_country_name(country_name):
    try:
        # Attempt to get the standard country name
        country = pycountry.countries.lookup(country_name)
        return country.name
    except LookupError:
        # Return the original name if no match is found
        return country_name

# Apply the function to the datasets
df_illness['standardized_country'] = df_illness['NAME'].apply(standardize_country_name)
df_contnent['standardized_country'] = df_contnent['country'].apply(standardize_country_name)
#drop replaced columns
